67      private CompilationLevel compilationLevel = CompilationLevel.SIMPLE_OPTIMIZATIONS;

70      private java.util.logging.Level loggingLevel = java.util.logging.Level.SEVERE;

76      private boolean customExternsOnly;

98      @Override
99      public String compress(String source) {
100 
101         StringWriter writer = new StringWriter();
102 
103         // prepare source
104         List<SourceFile> input = new ArrayList<>();
105         input.add(SourceFile.fromCode("source.js", source));
106 
107         // prepare externs
108         List<SourceFile> externsList = new ArrayList<>();
109         if (compilationLevel.equals(CompilationLevel.ADVANCED_OPTIMIZATIONS)) {
110             // default externs
111             if (!customExternsOnly) {
112                 try {
113                     externsList = getDefaultExterns();
114                 } catch (IOException e) {
115                     logger.error("", e);
116                 }
117             }
118             // add user defined externs
119             if (externs != null) {
120                 for (SourceFile extern : externs) {
121                     externsList.add(extern);
122                 }
123             }
124             // add empty externs
125             if (externsList.isEmpty()) {
126                 externsList.add(SourceFile.fromCode("externs.js", ""));
127             }
128         } else {
129             // empty externs
130             externsList.add(SourceFile.fromCode("externs.js", ""));
131         }
132 
133         Compiler.setLoggingLevel(loggingLevel);
134 
135         Compiler compiler = new Compiler();
136         compiler.disableThreads();
137 
138         compilationLevel.setOptionsForCompilationLevel(compilerOptions);
139         warningLevel.setOptionsForWarningLevel(compilerOptions);
140 
141         Result result = compiler.compile(externsList, input, compilerOptions);
142 
143         if (result.success) {
144             writer.write(compiler.toSource());
145         } else {
146             writer.write(source);
147         }
148 
149         return writer.toString();
150 
151     }

153     /**
154      * Gets the default externs.
155      *
156      * @return the default externs
157      *
158      * @throws IOException
159      *             Signals that an I/O exception has occurred.
160      */
161     // read default externs from closure.jar
162     private List<SourceFile> getDefaultExterns() throws IOException {
163         InputStream input = ClosureJavaScriptCompressor.class.getResourceAsStream("/externs.zip");
164         List<SourceFile> externList = new ArrayList<>();
165         try (ZipInputStream zip = new ZipInputStream(input)) {
166             for (ZipEntry entry; (entry = zip.getNextEntry()) != null;) {
167                 externList.add(SourceFile.builder().withCharset(Charset.defaultCharset())
168                         .withContent(ByteStreams.limit(zip, entry.getSize())).withPath(entry.getName()).build());
169             }
170         }
171         return externList;
172     }

186     /**

196      *
197      * @see <a href="http://code.google.com/closure/compiler/docs/api-tutorial3.html">Advanced Compilation and
198      *      Externs</a>

200      *      Levels</a>
