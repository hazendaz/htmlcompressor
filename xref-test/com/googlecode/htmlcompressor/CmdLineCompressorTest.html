75      @Test
76      void testCompressHtmlFileToOutputFile() throws IOException {
77          Path input = tmpDir.resolve("input.html");
78          Path output = tmpDir.resolve("output.html");
79          Files.writeString(input, SAMPLE_HTML);
80  
81          CmdLineCompressor c = new CmdLineCompressor(new String[] { "-o", output.toString(), input.toString() });
82          c.process();
83  
84          String result = Files.readString(output);

203     @Test
204     void testCompressHtmlWithAdvancedFlags() throws IOException {
205         Path input = tmpDir.resolve("in.html");
206         Path output = tmpDir.resolve("out.html");
207         Files.writeString(input,
208                 "<!doctype html PUBLIC \"-//W3C//DTD HTML 4.01//EN\" \"http://www.w3.org/TR/html4/strict.dtd\">"
209                         + "<html><body>" + "<script language=\"javascript\" type=\"text/javascript\">var x=1;</script>"
210                         + "<link rel=\"stylesheet\" type=\"text/css\" href=\"a.css\">"
211                         + "<a href=\"javascript:void(0)\">click</a>" + "</body></html>");
212 
213         CmdLineCompressor c = new CmdLineCompressor(
214                 new String[] { "--simple-doctype", "--remove-script-attr", "--remove-link-attr", "--remove-js-protocol",
215                         "--simple-bool-attr", "-o", output.toString(), input.toString() });
216         c.process();
217 
218         String result = Files.readString(output);
219         assertTrue(result.contains("<!DOCTYPE html>"));
220         assertFalse(result.contains("type=\"text/javascript\""));
221     }
