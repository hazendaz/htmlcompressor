95      @Test
96      void testHtmlTagDoEndTagCompressesDefaultSettings() throws Exception {
97          StubJspWriter writer = new StubJspWriter();
98          StubBodyContent body = new StubBodyContent(writer, HTML_CONTENT);
99          StubPageContext ctx = new StubPageContext(writer);
100 
101         HtmlCompressorTag tag = new HtmlCompressorTag();
102         tag.setPageContext(ctx);
103         tag.setBodyContent(body);
104 
105         tag.doEndTag();
106 
107         String output = writer.getOutput();
108         assertNotNull(output);
109         // default removes comments and multi-spaces
110         assertFalse(output.contains("<!--"), "Expected HTML comment to be removed");
111         assertFalse(output.contains("  "), "Expected multiple spaces to be removed");
112         assertTrue(output.contains("<html>"));
113     }
